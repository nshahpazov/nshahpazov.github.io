--- title: "Maximum likelihood estimation with numeric implementation in Python" date: 2020-08-08T11:41:53+03:00 draft: true --- {{< highlight Python >}} import math def likelihood(dataset, theta): return sum([y*math.log(theta) - theta for y in dataset]) def bisect(dataset, start, end, n, cmp, precision=3): [left, right] = sorted([start, end], key=cmp) for _ in range(n): middle = (left + right) / 2 [left, right] = sorted([left, middle, right], key=cmp)[0:2] return round(middle, precision) # disect our bisect method dataset = [6, 5, 4, 6, 6, 3, 12, 7, 4, 2, 6, 7, 4] mle = bisect(dataset, 5, 6, n=13, cmp=lambda x: -likelihood(dataset, x)) print(mle) {{< / highlight >}}
