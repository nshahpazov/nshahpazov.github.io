{{< highlight Python >}} import math def likelihood(dataset, theta): return sum([y*math.log(theta) - theta for y in dataset]) {{< / highlight >}} {{< highlight Python >}} def bisect(dataset, start, end, n, cmp, precision=3): [left, right] = sorted([start, end], key=cmp) for _ in range(n): middle = (left + right) / 2 [left, right] = sorted([left, middle, right], key=cmp)[0:2] return round(middle, precision) {{< / highlight >}} {{< highlight Python >}} # disect our bisect method > data = [6, 5, 4, 6, 6, 3, 12, 7, 4, 2, 6, 7, 4] > mle = bisect(data, start=5, end=6, n=13, cmp=lambda x: -likelihood(data, x)) > print(mle) # prints 5.538 {{< / highlight >}}
